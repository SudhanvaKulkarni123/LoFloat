29// Stores the 4-bit integer value in the low four bits of a byte. The upper
30// four bits are left unspecified and ignored.
31template <typename UnderlyingTy>
32struct i4 {
33 private:
34 UnderlyingTy v_;
35
36 static_assert(
37 std::is_same_v<UnderlyingTy, uint8_t> ||
38 std::is_same_v<UnderlyingTy, int8_t>,
39 "The underyling type must be a signed or unsigned 8-bit integer.");
40
41 // Mask the upper four bits.
42 static inline constexpr UnderlyingTy Mask(UnderlyingTy v) { return v & 0x0F; }
43
44 // Mask the upper four bits and sign-extend for signed types.
45 static inline constexpr UnderlyingTy MaskAndSignExtend(UnderlyingTy v) {
46 return std::is_signed_v<UnderlyingTy> ? Mask(v) | ((v & 0x08) ? 0xF0 : 0x00)
47 : Mask(v);
48 }
49
50 // Casts to the corresponding UnderlyingTy value.
51 inline constexpr UnderlyingTy IntValue() const {
52 return MaskAndSignExtend(v_);
53 }
54
55 public:
56 constexpr i4() noexcept : v_(0) {}
57 constexpr i4(const i4& other) noexcept = default;
58 constexpr i4(i4&& other) noexcept = default;
59 constexpr i4& operator=(const i4& other) = default;
60 constexpr i4& operator=(i4&&) = default;
61
62 explicit constexpr i4(UnderlyingTy val) : v_(Mask(val)) {}
63 template <typename T>
64 explicit constexpr i4(T t) : i4(static_cast<UnderlyingTy>(t)) {}
65
66 static constexpr i4 lowest() {
67 return std::is_signed<UnderlyingTy>::value ? i4(-8) : i4(0);
68 }
69 static constexpr i4 highest() {
70 return std::is_signed<UnderlyingTy>::value ? i4(7) : i4(15);
71 }
72
73 template <typename T>
74 explicit constexpr operator T() const {
75 return static_cast<T>(IntValue());
76 }
77 // NOLINTNEXTLINE(google-explicit-constructor)
78 constexpr operator std::optional<int64_t>() const {
79 return static_cast<int64_t>(IntValue());
80 }
81
82 constexpr i4 operator-() const { return i4(-v_); }
83 constexpr i4 operator+(const i4& other) const { return i4(v_ + other.v_); }
84 constexpr i4 operator-(const i4& other) const { return i4(v_ - other.v_); }
85 constexpr i4 operator*(const i4& other) const { return i4(v_ * other.v_); }
86 constexpr i4 operator/(const i4& other) const {
87 return i4(IntValue() / other.IntValue());
88 }
89 constexpr i4 operator%(const i4& other) const {
90 return i4((IntValue() % other.IntValue()));
91 }
92
93 constexpr i4 operator&(const i4& other) const { return i4(v_ & other.v_); }
94 constexpr i4 operator|(const i4& other) const { return i4(v_ | other.v_); }
95 constexpr i4 operator^(const i4& other) const { return i4(v_ ^ other.v_); }
96 constexpr i4 operator~() const { return i4(~v_); }
97 constexpr i4 operator>>(int amount) const { return i4(IntValue() >> amount); }
98 constexpr i4 operator<<(int amount) const { return i4(v_ << amount); }
99
100 constexpr bool operator==(const i4& other) const {
101 return Mask(v_) == Mask(other.v_);
102 }
103 constexpr bool operator!=(const i4& other) const {
104 return Mask(v_) != Mask(other.v_);
105 }
106 constexpr bool operator<(const i4& other) const {
107 return IntValue() < other.IntValue();
108 }
109 constexpr bool operator>(const i4& other) const {
110 return IntValue() > other.IntValue();
111 }
112 constexpr bool operator<=(const i4& other) const {
113 return IntValue() <= other.IntValue();
114 }
115 constexpr bool operator>=(const i4& other) const {
116 return IntValue() >= other.IntValue();
117 }
118
119 constexpr bool operator==(int64_t other) const { return IntValue() == other; }
120 constexpr bool operator!=(int64_t other) const { return IntValue() != other; }
121 constexpr bool operator<(int64_t other) const { return IntValue() < other; }
122 constexpr bool operator>(int64_t other) const { return IntValue() > other; }
123 constexpr bool operator<=(int64_t other) const { return IntValue() <= other; }
124 constexpr bool operator>=(int64_t other) const { return IntValue() >= other; }
125
126 friend constexpr bool operator==(int64_t a, const i4& b) {
127 return a == b.IntValue();
128 }
129 friend constexpr bool operator!=(int64_t a, const i4& b) {
130 return a != b.IntValue();
131 }
132 friend constexpr bool operator<(int64_t a, const i4& b) {
133 return a < b.IntValue();
134 }
135 friend constexpr bool operator>(int64_t a, const i4& b) {
136 return a > b.IntValue();
137 }
138 friend constexpr bool operator<=(int64_t a, const i4& b) {
139 return a <= b.IntValue();
140 }
141 friend constexpr bool operator>=(int64_t a, const i4& b) {
142 return a >= b.IntValue();
143 }
144
145 constexpr i4& operator++() {
146 v_ = Mask(v_ + 1);
147 return *this;
148 }
149
150 constexpr i4 operator++(int) {
151 i4 orig = *this;
152 this->operator++();
153 return orig;
154 }
155
156 constexpr i4& operator--() {
157 v_ = Mask(v_ - 1);
158 return *this;
159 }
160
161 constexpr i4 operator--(int) {
162 i4 orig = *this;
163 this->operator--();
164 return orig;
165 }
166
167 constexpr i4& operator+=(const i4& other) {
168 *this = *this + other;
169 return *this;
170 }
171 constexpr i4& operator-=(const i4& other) {
172 *this = *this - other;
173 return *this;
174 }
175 constexpr i4& operator*=(const i4& other) {
176 *this = *this * other;
177 return *this;
178 }
179 constexpr i4& operator/=(const i4& other) {
180 *this = *this / other;
181 return *this;
182 }
183 constexpr i4& operator%=(const i4& other) {
184 *this = *this % other;
185 return *this;
186 }
187 constexpr i4& operator&=(const i4& other) {
188 *this = *this & other;
189 return *this;
190 }
191 constexpr i4& operator|=(const i4& other) {
192 *this = *this | other;
193 return *this;
194 }
195 constexpr i4& operator^=(const i4& other) {
196 *this = *this ^ other;
197 return *this;
198 }
199 constexpr i4& operator>>=(int amount) {
200 *this = *this >> amount;
201 return *this;
202 }
203 constexpr i4& operator<<=(int amount) {
204 *this = *this << amount;
205 return *this;
206 }
207
208 friend ::std::ostream& operator<<(::std::ostream& os, const i4& num) {
209 os << static_cast<int16_t>(num);
210 return os;
211 }
212
213 std::string ToString() const {
214 std::ostringstream os;
215 os << static_cast<int16_t>(*this);
216 return os.str();
217 }
218};
