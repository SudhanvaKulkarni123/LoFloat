209struct SingleInfChecker {
210 bool operator()(uint32_t bits) const {
211 static constexpr uint32_t ExponentMask = 0x7F800000;
212 static constexpr uint32_t FractionMask = 0x007FFFFF;
213 // Infinity => exponent=0xFF, fraction=0
214 bool isInf = ((bits & ExponentMask) == ExponentMask) &&
215 ((bits & FractionMask) == 0);
216 return isInf;
217 }
218
219 uint32_t infBitPattern() const {
220 // +∞ => 0x7F800000
221 return 0x7F800000;
222 }

226 return 0xFF800000;
227 }
228
229 uint32_t minPosInf() const {
230 // +∞ => 0x7F800000
231 return 0x7F800000;
232 }

241struct SingleNaNChecker {
